```python
def classify_intent_node(state: AgentState, llm: LLMClient) -> AgentState:
    intent = llm.classify_intent(state.get("user_input", ""))
    if intent not in {"ticket", "build", "docs", "unknown"}:
        intent = "unknown"
    return update_state(
        state,
        intent=intent,
        ticket_id=extract_ticket_id(state.get("user_input", "")),
        build_id=extract_build_id(state.get("user_input", "")),
    )
```
