Substitua a função `final_answer_node` por:

```python
def final_answer_node(state: AgentState) -> AgentState:
    intent = state.get("intent")
    context = state.get("context") or {}

    if state.get("error") and not context:
        return update_state(state, final_answer=f"Nao consegui responder com seguranca: {state['error']}")

    if intent == "ticket":
        answer = (
            f"Resumo: ticket {context.get('id')} - {context.get('summary')}\n"
            f"Evidencias: severidade {context.get('severity')}, servico {context.get('service')}, "
            f"status {context.get('status')}, build relacionado {context.get('build_id')}.\n"
            "Proximo passo: verificar build relacionado e healthcheck do servico."
        )
        return update_state(state, final_answer=answer)

    if intent == "build":
        answer = (
            f"Resumo: build {context.get('id')} esta com status {context.get('status')}.\n"
            f"Evidencias: servico {context.get('service')}, branch {context.get('branch')}, "
            f"etapa falha {context.get('failed_step')}, log {context.get('log_excerpt')}.\n"
            "Proximo passo: revisar a etapa falha e relacionar com tickets recentes."
        )
        return update_state(state, final_answer=answer)

    if intent == "docs":
        resource = context.get("resource", {})
        prompt = context.get("prompt", {})
        answer = (
            f"Resumo: consulte {resource.get('title')} para orientar a triagem.\n"
            f"Evidencias: {resource.get('content')}\n"
            f"Proximo passo: {prompt.get('content')}"
        )
        return update_state(state, final_answer=answer)

    return update_state(
        state,
        final_answer="Nao consegui identificar se a pergunta e sobre ticket, build ou documentacao.",
    )
```
